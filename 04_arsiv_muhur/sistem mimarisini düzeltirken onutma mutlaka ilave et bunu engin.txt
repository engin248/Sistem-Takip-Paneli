<input id="task" placeholder="görev">
<button onclick="createTask()">GÖNDER</button>

<div id="log"></div>

<script>

let DB = {
  tasks: []
};

// HER İŞLEM ADIM ADIM
const STEPS = ["plan","control1","exec","control2","master"];

// GÖREV OLUŞTUR
function createTask(){
  DB.tasks.push({
    id: Date.now(),
    content: document.getElementById("task").value,
    step: 0,
    history: [],
    error: false
  });
  render();
}

// HATA KONTROL (örnek simülasyon)
function hasError(){
  // %20 hata simülasyonu (gerçek sistemde kontrol mekanizması olacak)
  return Math.random() < 0.2;
}

// AJAN İŞLEMLERİ
function process(agent){

  DB.tasks.forEach(t => {

    // sadece doğru aşama çalışır
    if(STEPS[t.step] === agent){

      // HATA VARSA
      if(hasError()){
        t.history.push("HATA → SİLİNDİ → YENİDEN BAŞLADI");
        
        // TAM SİLME VE SIFIRLAMA
        t.step = 0;
        t.history.push("SIFIRDAN BAŞLADI");
        
        return;
      }

      // DOĞRU İŞLEM
      t.history.push(agent + " OK");
      t.step++;
    }

  });

  render();
}

// GÖRÜNTÜ
function render(){
  let out = "";

  DB.tasks.forEach(t => {
    out += `GÖREV: ${t.content}\n`;
    out += `DURUM: ${STEPS[t.step] || "TAMAMLANDI"}\n`;
    out += `AKIŞ:\n${t.history.join("\n")}\n`;
    out += "------------------------\n";
  });

  document.getElementById("log").innerText = out;
}

// 5 AŞAMA
setInterval(()=>process("plan"),1000);
setInterval(()=>process("control1"),1500);
setInterval(()=>process("exec"),2000);
setInterval(()=>process("control2"),2500);
setInterval(()=>process("master"),3000);

</script>
