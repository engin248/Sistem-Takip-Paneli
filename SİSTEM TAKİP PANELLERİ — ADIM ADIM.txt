3) SERVER (ÇEKİRDEK)

server.js

const express = require("express");
const fs = require("fs");

const app = express();
app.use(express.json());

const DB_FILE = "./db.json";

function readDB() {
  return JSON.parse(fs.readFileSync(DB_FILE));
}

function writeDB(data) {
  fs.writeFileSync(DB_FILE, JSON.stringify(data, null, 2));
}

// GÖREV OLUŞTUR
app.post("/task", (req, res) => {
  const db = readDB();

  const task = {
    id: Date.now(),
    content: req.body.content,
    step: 0,
    history: [],
    status: "active"
  };

  db.tasks.push(task);
  writeDB(db);

  res.json(task);
});

// GÖREVLERİ GETİR
app.get("/tasks", (req, res) => {
  const db = readDB();
  res.json(db.tasks);
});

// AJAN İŞLEMİ
const STEPS = ["plan","control1","exec","control2","master"];

app.post("/process", (req, res) => {
  const db = readDB();
  const { agent } = req.body;

  db.tasks.forEach(t => {
    if(t.status !== "active") return;

    if(STEPS[t.step] === agent){

      // HATA KONTROL (örnek)
      const error = Math.random() < 0.2;

      if(error){
        db.logs.push(`HATA: ${t.id} → SİLİNDİ`);
        
        // TAM SİLME
        t.step = 0;
        t.history = ["RESET"];
        return;
      }

      t.history.push(agent);
      t.step++;

      if(t.step >= STEPS.length){
        t.status = "done";
        db.logs.push(`TAMAMLANDI: ${t.id}`);
      }
    }
  });

  writeDB(db);
  res.json({ ok: true });
});

// LOG
app.get("/logs", (req, res) => {
  const db = readDB();
  res.json(db.logs);
});

app.listen(3000, "127.0.0.1", () => {
  console.log("SİSTEM BAŞLADI");
});

  await fetch("http://127.0.0.1:3000/task", {
    method:"POST",
    headers:{"Content-Type":"application/json"},
    body: JSON.stringify({
      content: document.getElementById("task").value
    })

async function load(){
  const t = await fetch("http://127.0.0.1:3000/tasks").then(r=>r.json());
  const l = await fetch("http://127.0.0.1:3000/logs").then(r=>r.json());

  let out = "TASKS:\n";
  t.forEach(x=>{
    out += `${x.id} | step:${x.step} | ${x.status}\n`;
    out += x.history.join("->") + "\n\n";
  });

  out += "\nLOGS:\n";
  l.forEach(x=> out+=x+"\n");

  document.getElementById("out").innerText = out;
}

server.js (GÜÇLENDİRİLMİŞ)
const express = require("express");
const fs = require("fs");

const app = express();
app.use(express.json());

const DB_FILE = "./db.json";
let LOCK = false;

const STEPS = ["plan","control1","exec","control2","master"];

function readDB(){
  return JSON.parse(fs.readFileSync(DB_FILE));
}

function writeDB(data){
  const temp = DB_FILE + ".tmp";
  fs.writeFileSync(temp, JSON.stringify(data, null, 2));
  fs.renameSync(temp, DB_FILE); // ATOMİK YAZIM
}

// LOCK
function acquireLock(){
  if(LOCK) return false;
  LOCK = true;
  return true;
}

function releaseLock(){
  LOCK = false;
}

// VALIDATE
function validate(task, step){
  if(step === "plan") return task.content.length > 2;
  if(step === "control1") return task.history.includes("plan");
  if(step === "exec") return task.history.includes("control1");
  if(step === "control2") return task.history.includes("exec");
  if(step === "master") return task.history.length === 4;
  return false;
}

// TASK
app.post("/task",(req,res)=>{
  if(!acquireLock()) return res.json({busy:true});

  const db = readDB();

  const task = {
    id: Date.now(),
    content: req.body.content,
    step: 0,
    history: [],
    status: "active",
    version: 0
  };

  db.tasks.push(task);
  db.logs.push("CREATE " + task.id);

  writeDB(db);
  releaseLock();

  res.json(task);
});

// PROCESS
app.post("/process",(req,res)=>{
  if(!acquireLock()) return res.json({busy:true});

  const db = readDB();
  const {agent} = req.body;

  db.tasks.forEach(t=>{

    if(t.status !== "active") return;

    const expected = STEPS[t.step];
    if(agent !== expected) return;

    const ok = validate(t, agent);

    if(!ok){
      db.logs.push("RESET " + t.id);

      // TAM RESET
      t.step = 0;
      t.history = [];
      t.version++;
      return;
    }

    t.history.push(agent);
    t.step++;
    t.version++;

    if(t.step >= STEPS.length){
      t.status = "done";
      db.logs.push("DONE " + t.id);
    }

  });

  writeDB(db);
  releaseLock();

  res.json({ok:true});
});

app.get("/tasks",(req,res)=>res.json(readDB().tasks));
app.get("/logs",(req,res)=>res.json(readDB().logs));

app.listen(3000,"127.0.0.1");
